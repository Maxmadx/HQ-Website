// src/hooks/useAdmin.js
import { useState, useEffect } from 'react';
import { onAuthStateChanged, signInWithEmailAndPassword, signOut } from 'firebase/auth';
import { auth } from '../lib/firebase.js';

export function useAdmin() {
  const [user, setUser]       = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError]     = useState(null);

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, (u) => {
      setUser(u);
      setLoading(false);
    });
    return unsubscribe;
  }, []);

  async function login(email, password) {
    setError(null);
    try {
      await signInWithEmailAndPassword(auth, email, password);
    } catch (e) {
      setError(e.message);
    }
  }

  async function logout() {
    await signOut(auth);
  }

  return { user, loading, error, login, logout };
}
```

import { Navigate } from 'react-router-dom';
import { useAdmin } from '../../hooks/useAdmin.js';

export default function AdminRoute({ children }) {
  const { user, loading } = useAdmin();

  if (loading) {
    return (
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: '100vh', fontFamily: 'monospace', fontSize: '0.8rem', color: '#999' }}>

      </div>

  }

  return children;
}

// src/pages/admin/AdminLogin.jsx
import { useState } from 'react';
import { Navigate } from 'react-router-dom';
import { useAdmin } from '../../hooks/useAdmin.js';
import './admin.css';

export default function AdminLogin() {
  const { user, loading, error, login } = useAdmin();
  const [email, setEmail]       = useState('');
  const [password, setPassword] = useState('');
  const [submitting, setSubmitting] = useState(false);

  if (loading) return null;
  if (user) return <Navigate to="/admin" replace />;

  async function handleSubmit(e) {
    e.preventDefault();
    setSubmitting(true);
    await login(email, password);
    setSubmitting(false);
  }

  return (
    <div style={{ minHeight: '100vh', display: 'flex', alignItems: 'center', justifyContent: 'center', background: '#f4f4f5', fontFamily: "'Space Grotesk', sans-serif" }}>
      <div style={{ background: '#fff', border: '1px solid #e4e4e7', borderRadius: 10, padding: '2.5rem 2rem', width: '100%', maxWidth: 380 }}>
        <div style={{ marginBottom: '2rem' }}>
          <p style={{ fontFamily: 'monospace', fontSize: '0.6rem', letterSpacing: '0.2em', textTransform: 'uppercase', color: '#71717a', marginBottom: '0.4rem' }}>HQ Aviation</p>
          <h1 style={{ fontSize: '1.3rem', fontWeight: 700, color: '#1a1a1a', margin: 0 }}>Admin Login</h1>
        </div>

        <form onSubmit={handleSubmit} className="adm-form">
          <div className="adm-field">
            <label className="adm-label">Email</label>
            <input className="adm-input" type="email" value={email} onChange={e => setEmail(e.target.value)} required autoFocus />
          </div>
          <div className="adm-field">
            <label className="adm-label">Password</label>
            <input className="adm-input" type="password" value={password} onChange={e => setPassword(e.target.value)} required />
          </div>
          {error && <p style={{ fontSize: '0.75rem', color: '#ef4444', margin: 0 }}>{error}</p>}
          <button className="adm-btn adm-btn--primary" type="submit" disabled={submitting} style={{ marginTop: '0.5rem' }}>
            {submitting ? 'Signing in…' : 'Sign in'}
          </button>
        </form>
      </div>
    </div>
  );
}
```

### Task 7: Firestore security rules — admin only

**Files:** Modify `firestore.rules`

- [ ] **Step 1: Update Firestore rules**

Replace the contents of `firestore.rules`:

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: is the request from a signed-in admin?
    function isAdmin() {
      return request.auth != null && request.auth.token.email == 'REPLACE_WITH_ADMIN_EMAIL';
    }

    // Public read for content collections
    match /listings/{id}    { allow read: if true; allow write: if isAdmin(); }
    match /blog_posts/{id}  { allow read: if true; allow write: if isAdmin(); }
    match /pricing/{id}     { allow read: if true; allow write: if isAdmin(); }
    match /reviews/{id}     { allow read: if true; allow write: if isAdmin(); }
    match /image_slots/{id} { allow read: if true; allow write: if isAdmin(); }

    // Leads: public write (form submissions), admin read
    match /leads/{id}       { allow create: if true; allow read, update, delete: if isAdmin(); }

    // Wall of cool: public create (upload), admin read/update/delete
    match /wall_of_cool/{id} { allow create: if true; allow read, update, delete: if isAdmin(); }

    // Analytics: public create, admin read
    match /page_events/{id}  { allow create: if true; allow read, delete: if isAdmin(); }

    // Default deny
    match /{document=**} { allow read, write: if false; }
  }
}
```

Replace `REPLACE_WITH_ADMIN_EMAIL` with the actual admin email (e.g. `owner@hqaviation.co.uk`).
